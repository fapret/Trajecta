var spanTags = document.querySelectorAll('span.ui-treenode-content');
var spanTexts = [];

// Extract text content
for (var i = 0; i < spanTags.length; i++) {
    spanTexts.push(spanTags[i].textContent);
}

// Create a Blob containing the text data
var textBlob = new Blob([spanTexts.join('\n')], { type: 'text/plain' });

// Create a temporary URL for the Blob
var url = URL.createObjectURL(textBlob);

// Create a link element to trigger the download
var a = document.createElement('a');
a.href = url;
a.download = 'spanTexts.txt'; // Specify the filename

// Trigger a click event to download the file
a.click();

// Clean up by revoking the Blob URL
URL.revokeObjectURL(url);
